# Calendar Flip

A top-edge constrained flip works well for date tiles and calendars.

import { PeelWrapper, PeelTop, PeelBack, PeelBottom } from "react-peel";

## Implementation

```tsx
import React, { useState } from "react";
import { PeelWrapper, PeelTop, PeelBack, PeelBottom, usePeel } from "react-peel";

export function CalendarFlip() {
  const [date, setDate] = useState(new Date());
  const { peelRef, reset } = usePeel();

  const nextDate = new Date(date);
  nextDate.setDate(date.getDate() + 1);

  return (
    <PeelWrapper
      ref={peelRef}
      preset="calendar"
      width={200}
      height={180}
      drag
      onPeelProgress={(progress) => {
        if (progress > 0.6) {
          setDate(nextDate);
          reset();
        }
      }}
    >
      <PeelTop>{date.getDate()}</PeelTop>
      <PeelBack style={{ background: "#f5f5f5" }} />
      <PeelBottom>{nextDate.getDate()}</PeelBottom>
    </PeelWrapper>
  );
}
```
