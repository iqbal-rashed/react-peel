# Envelope

An envelope opening animation with letter reveal. Perfect for email interfaces, invitations, or message notifications.

## Preview

An envelope that opens to reveal a letter inside with smooth animation.

## Implementation

```tsx
import React, { useState } from "react";
import { PeelWrapper, PeelTop, PeelBack, PeelBottom, usePeel } from "react-peel";

interface EnvelopeProps {
  subject: string;
  message: string;
  sender: string;
  date?: string;
  onOpen?: () => void;
}

export function Envelope({
  subject,
  message,
  sender,
  date = new Date().toLocaleDateString(),
  onOpen,
}: EnvelopeProps) {
  const { peelRef, animate, reset } = usePeel();
  const [isOpen, setIsOpen] = useState(false);

  const handleOpen = async () => {
    if (isOpen) return;

    await animate({
      to: { x: 150, y: -50 },
      duration: 600,
      easing: "easeOut",
    });

    setIsOpen(true);
    onOpen?.();
  };

  const handleClose = async () => {
    reset();
    setIsOpen(false);
  };

  return (
    <div
      style={{
        width: 320,
        perspective: "1000px",
      }}
    >
      {/* Envelope */}
      <PeelWrapper
        ref={peelRef}
        preset="envelope"
        width={320}
        height={200}
        corner="TOP_RIGHT"
        constraints={["TOP_LEFT"]}
        drag
        onPeelProgress={(progress) => {
          if (progress > 0.5 && !isOpen) {
            setIsOpen(true);
            onOpen?.();
          }
        }}
      >
        {/* Flap */}
        <PeelTop
          style={{
            background: "#e8d5b7",
            clipPath: "polygon(0 0, 50% 60%, 100% 0)",
            borderBottom: "2px solid #d4c4a8",
          }}
        />
        <PeelBack
          style={{
            background: "#f5e6d3",
            clipPath: "polygon(0 0, 50% 60%, 100% 0)",
          }}
        />
        {/* Envelope body with letter preview */}
        <PeelBottom
          style={{
            background: "#e8d5b7",
            display: "flex",
            alignItems: "center",
            justifyContent: "center",
            padding: "20px",
          }}
        >
          {/* Letter inside */}
          <div
            style={{
              width: "90%",
              height: "80%",
              background: "#fff",
              boxShadow: "0 2px 8px rgba(0,0,0,0.1)",
              padding: "16px",
              overflow: "hidden",
            }}
          >
            <div
              style={{
                fontSize: "10px",
                color: "#999",
                marginBottom: "8px",
              }}
            >
              {date}
            </div>
            <div
              style={{
                fontSize: "14px",
                fontWeight: "bold",
                marginBottom: "8px",
                color: "#333",
              }}
            >
              {subject}
            </div>
            <div
              style={{
                fontSize: "12px",
                color: "#666",
                lineHeight: 1.4,
              }}
            >
              {message.substring(0, 80)}...
            </div>
          </div>
        </PeelBottom>
      </PeelWrapper>

      {/* Envelope base */}
      <div
        style={{
          marginTop: "-100px",
          width: 320,
          height: 100,
          background: "#e8d5b7",
          borderBottomLeftRadius: "8px",
          borderBottomRightRadius: "8px",
          boxShadow: "0 4px 12px rgba(0,0,0,0.15)",
        }}
      />

      {/* Full letter view when open */}
      {isOpen && (
        <div
          style={{
            marginTop: "20px",
            background: "#fff",
            padding: "24px",
            borderRadius: "8px",
            boxShadow: "0 4px 16px rgba(0,0,0,0.1)",
            animation: "slideUp 0.3s ease-out",
          }}
        >
          <div
            style={{
              display: "flex",
              justifyContent: "space-between",
              marginBottom: "16px",
              paddingBottom: "16px",
              borderBottom: "1px solid #eee",
            }}
          >
            <div>
              <div style={{ fontWeight: "bold", color: "#333" }}>{sender}</div>
              <div style={{ fontSize: "14px", color: "#666" }}>{subject}</div>
            </div>
            <div style={{ fontSize: "14px", color: "#999" }}>{date}</div>
          </div>
          <div
            style={{
              fontSize: "16px",
              lineHeight: 1.6,
              color: "#333",
              whiteSpace: "pre-wrap",
            }}
          >
            {message}
          </div>
          <button
            onClick={handleClose}
            style={{
              marginTop: "20px",
              padding: "12px 24px",
              background: "#f5f5f5",
              border: "none",
              borderRadius: "8px",
              cursor: "pointer",
            }}
          >
            Close Letter
          </button>
        </div>
      )}

      {/* Open button */}
      {!isOpen && (
        <button
          onClick={handleOpen}
          style={{
            display: "block",
            width: "100%",
            marginTop: "16px",
            padding: "12px",
            background: "#8b7355",
            color: "#fff",
            border: "none",
            borderRadius: "8px",
            fontSize: "16px",
            cursor: "pointer",
          }}
        >
          ‚úâÔ∏è Open Envelope
        </button>
      )}

      <style>{`
        @keyframes slideUp {
          from { opacity: 0; transform: translateY(20px); }
          to { opacity: 1; transform: translateY(0); }
        }
      `}</style>
    </div>
  );
}
```

## Inbox Example

```tsx
function Inbox() {
  const emails = [
    {
      subject: "Welcome to the team!",
      sender: "HR Department",
      message: "We're excited to have you join us...",
    },
    {
      subject: "Meeting Tomorrow",
      sender: "John Smith",
      message: "Hi, just a reminder about our meeting...",
    },
  ];

  return (
    <div style={{ display: "flex", flexDirection: "column", gap: "24px" }}>
      {emails.map((email, i) => (
        <Envelope key={i} {...email} />
      ))}
    </div>
  );
}
```
