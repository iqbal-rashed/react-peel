import { PeelWrapper, PeelTop, PeelBack, PeelBottom } from "react-peel";

## Implementation

```tsx
import { PeelWrapper, PeelTop, PeelBack, PeelBottom } from "react-peel";

interface EnvelopeProps {
  message: string;
}

export function Envelope({ message }: EnvelopeProps) {
  return (
    <PeelWrapper
      preset="envelope"
      width={320}
      height={200}
      corner="TOP_RIGHT"
      constraints={["TOP_LEFT"]}
      drag
    >
      <PeelTop
        style={{
          background: "#e8d5b7",
          clipPath: "polygon(0 0, 50% 60%, 100% 0)",
          borderBottom: "2px solid #d4c4a8",
        }}
      />
      <PeelBack
        style={{
          background: "#f5e6d3",
          clipPath: "polygon(0 0, 50% 60%, 100% 0)",
        }}
      />
      <PeelBottom
        style={{
          background: "#e8d5b7",
          display: "flex",
          alignItems: "center",
          justifyContent: "center",
          padding: 20,
        }}
      >
        <div style={{ background: "#fff", padding: 16, width: "90%" }}>
          {message}
        </div>
      </PeelBottom>
    </PeelWrapper>
  );
}
```

## Usage

```tsx
<Envelope message="You are invited to the event." />
```
