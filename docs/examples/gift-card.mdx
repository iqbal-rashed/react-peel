# Gift Card

Reveal a code with a dramatic peel and a call to action.

import { PeelWrapper, PeelTop, PeelBack, PeelBottom } from "react-peel";

## Implementation

```tsx
import React, { useState } from "react";
import { PeelWrapper, PeelTop, PeelBack, PeelBottom } from "react-peel";

interface GiftCardProps {
  amount: number;
  code: string;
}

export function GiftCard({ amount, code }: GiftCardProps) {
  const [revealed, setRevealed] = useState(false);

  return (
    <PeelWrapper
      preset="giftCard"
      width={320}
      height={200}
      drag
      onPeelProgress={(progress) => {
        if (!revealed && progress > 0.6) setRevealed(true);
      }}
    >
      <PeelTop style={{ background: "#667eea", color: "#fff", padding: 20 }}>
        <div style={{ fontSize: 14 }}>GIFT CARD</div>
        <div style={{ fontSize: 36, fontWeight: "bold" }}>${amount}</div>
        <div style={{ marginTop: 12 }}>Peel to reveal</div>
      </PeelTop>
      <PeelBack style={{ background: "#5e60ce" }} />
      <PeelBottom
        style={{
          background: "#fff",
          display: "flex",
          alignItems: "center",
          justifyContent: "center",
          fontSize: 22,
          fontWeight: "bold",
          color: "#5e60ce",
        }}
      >
        {revealed ? code : ""}
      </PeelBottom>
    </PeelWrapper>
  );
}
```

## Usage

```tsx
<GiftCard amount={50} code="GIFT50" />
```
