# Magazine

Create a page flip experience for editorial content.

import { PeelWrapper, PeelTop, PeelBack, PeelBottom } from "react-peel";

## Live Demo

<div style={{ padding: "20px", display: "flex", justifyContent: "center" }}>
  <PeelWrapper preset="pageFlip" width={240} height={320} drag>
    <PeelTop style={{ background: "#fff9c4", padding: 16 }}>
      <h3 style={{ margin: 0 }}>Feature Story</h3>
      <p style={{ margin: "8px 0 0" }}>Peel the corner to turn the page.</p>
    </PeelTop>
    <PeelBack style={{ background: "#f5f5f5" }} />
    <PeelBottom style={{ background: "#fafafa", padding: 16 }}>
      <h3 style={{ margin: 0 }}>Next Page</h3>
      <p style={{ margin: "8px 0 0" }}>More content here.</p>
    </PeelBottom>
  </PeelWrapper>
</div>

## Implementation

```tsx
import React, { useState } from "react";
import { PeelWrapper, PeelTop, PeelBack, PeelBottom, usePeel } from "react-peel";

interface MagazineProps {
  pages: React.ReactNode[];
}

export function Magazine({ pages }: MagazineProps) {
  const [index, setIndex] = useState(0);
  const { peelRef, reset } = usePeel();

  const current = pages[index];
  const next = pages[index + 1];

  return (
    <PeelWrapper
      ref={peelRef}
      preset="pageFlip"
      width={260}
      height={340}
      drag
      onPeelProgress={(progress) => {
        if (progress > 0.6 && index < pages.length - 1) {
          setIndex((i) => i + 1);
          reset();
        }
      }}
    >
      <PeelTop style={{ background: "#fff", padding: 16 }}>{current}</PeelTop>
      <PeelBack style={{ background: "#f5f5f5" }} />
      <PeelBottom style={{ background: "#fafafa", padding: 16 }}>
        {next || "End of issue"}
      </PeelBottom>
    </PeelWrapper>
  );
}
```

## Usage

```tsx
<Magazine
  pages={[
    <div key="one">Cover Story</div>,
    <div key="two">Feature Article</div>,
    <div key="three">Photo Spread</div>,
  ]}
/>
```
