# Photo Album

Use the `photoAlbum` preset to build a page-flip gallery.

import { PeelWrapper, PeelTop, PeelBack, PeelBottom } from "react-peel";

## Live Demo

<div style={{ padding: "20px", display: "flex", justifyContent: "center" }}>
  <PeelWrapper preset="photoAlbum" width={240} height={320} drag>
    <PeelTop style={{ background: "#cfd8dc", display: "flex", alignItems: "center", justifyContent: "center" }}>
      Current Photo
    </PeelTop>
    <PeelBack style={{ background: "#b0bec5" }} />
    <PeelBottom style={{ background: "#eceff1", display: "flex", alignItems: "center", justifyContent: "center" }}>
      Next Photo
    </PeelBottom>
  </PeelWrapper>
</div>

## Implementation

```tsx
import React, { useState } from "react";
import { PeelWrapper, PeelTop, PeelBack, PeelBottom, usePeel } from "react-peel";

interface PhotoAlbumProps {
  photos: string[];
}

export function PhotoAlbum({ photos }: PhotoAlbumProps) {
  const [index, setIndex] = useState(0);
  const { peelRef, reset } = usePeel();

  const current = photos[index];
  const next = photos[index + 1];

  return (
    <PeelWrapper
      ref={peelRef}
      preset="photoAlbum"
      width={260}
      height={340}
      drag
      onPeelProgress={(progress) => {
        if (progress > 0.8 && index < photos.length - 1) {
          setIndex((i) => i + 1);
          reset();
        }
      }}
    >
      <PeelTop style={{ backgroundImage: `url(${current})`, backgroundSize: "cover" }} />
      <PeelBack style={{ backgroundImage: `url(${current})`, backgroundSize: "cover" }} />
      <PeelBottom
        style={{
          backgroundImage: next ? `url(${next})` : undefined,
          backgroundColor: next ? undefined : "#eceff1",
          backgroundSize: "cover",
        }}
      />
    </PeelWrapper>
  );
}
```

## Usage

```tsx
<PhotoAlbum photos={["/photos/one.jpg", "/photos/two.jpg", "/photos/three.jpg"]} />
```
