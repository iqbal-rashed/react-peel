# Product Card

Reveal a discount code by peeling the product image.

import { PeelWrapper, PeelTop, PeelBack, PeelBottom } from "react-peel";

## Implementation

```tsx
import React, { useState } from "react";
import { PeelWrapper, PeelTop, PeelBack, PeelBottom } from "react-peel";

interface ProductCardProps {
  name: string;
  price: string;
  discountCode: string;
}

export function ProductCard({ name, price, discountCode }: ProductCardProps) {
  const [revealed, setRevealed] = useState(false);

  return (
    <div style={{ width: 280, borderRadius: 12, overflow: "hidden" }}>
      <PeelWrapper
        preset="revealCard"
        width={280}
        height={180}
        drag
        onPeelProgress={(progress) => {
          if (!revealed && progress > 0.5) {
            setRevealed(true);
          }
        }}
      >
        <PeelTop style={{ background: "#fafafa" }}>Product Image</PeelTop>
        <PeelBack style={{ background: "#ff7043" }} />
        <PeelBottom
          style={{
            background: "#2e7d32",
            color: "#fff",
            display: "flex",
            alignItems: "center",
            justifyContent: "center",
            fontWeight: "bold",
          }}
        >
          {discountCode}
        </PeelBottom>
      </PeelWrapper>
      <div style={{ padding: 16 }}>
        <div style={{ fontWeight: "bold" }}>{name}</div>
        <div style={{ color: "#555", marginTop: 4 }}>{price}</div>
        {revealed && (
          <div style={{ marginTop: 8, color: "#2e7d32" }}>
            Discount revealed
          </div>
        )}
      </div>
    </div>
  );
}
```

## Usage

```tsx
<ProductCard name="Premium Headphones" price="$199.99" discountCode="SAVE20" />
```
