# Scratch Card

A peel reveal works like a scratch-off card. Use progress callbacks to trigger rewards.

import { PeelWrapper, PeelTop, PeelBack, PeelBottom } from "react-peel";

## Implementation

```tsx
import React, { useState } from "react";
import { PeelWrapper, PeelTop, PeelBack, PeelBottom } from "react-peel";

interface ScratchCardProps {
  prize: string;
  revealThreshold?: number;
}

export function ScratchCard({ prize, revealThreshold = 0.6 }: ScratchCardProps) {
  const [revealed, setRevealed] = useState(false);

  return (
    <PeelWrapper
      preset="scratchCard"
      width={280}
      height={140}
      drag
      fadeThreshold={0.85}
      onPeelProgress={(progress) => {
        if (!revealed && progress >= revealThreshold) {
          setRevealed(true);
        }
      }}
    >
      <PeelTop
        style={{
          background: "linear-gradient(135deg, #9e9e9e 0%, #757575 100%)",
          display: "flex",
          alignItems: "center",
          justifyContent: "center",
          color: "#fff",
        }}
      >
        {revealed ? "" : "Scratch to reveal"}
      </PeelTop>
      <PeelBack
        style={{
          background: "linear-gradient(135deg, #bdbdbd 0%, #9e9e9e 100%)",
        }}
      />
      <PeelBottom
        style={{
          background: "linear-gradient(135deg, #4caf50 0%, #2e7d32 100%)",
          display: "flex",
          alignItems: "center",
          justifyContent: "center",
          color: "#fff",
          fontSize: 24,
          fontWeight: "bold",
        }}
      >
        {prize}
      </PeelBottom>
    </PeelWrapper>
  );
}
```

## Usage

```tsx
<ScratchCard prize="25% OFF" />
```
