# Sticky Note

A realistic sticky note with a peelable corner effect. Perfect for reminders, notifications, or decorative UI elements.

## Preview

A yellow sticky note that users can peel back to reveal a hidden message or simply interact with.

## Implementation

```tsx
import React, { useState } from "react";
import { PeelWrapper, PeelTop, PeelBack, PeelBottom, usePeel, peelHint } from "react-peel";

interface StickyNoteProps {
  message: string;
  hiddenMessage?: string;
  color?: string;
  width?: number;
  height?: number;
  showHint?: boolean;
}

export function StickyNote({
  message,
  hiddenMessage = "You found the secret!",
  color = "#fff9c4",
  width = 200,
  height = 200,
  showHint = true,
}: StickyNoteProps) {
  const { peelRef } = usePeel();
  const [hasInteracted, setHasInteracted] = useState(false);

  React.useEffect(() => {
    if (showHint && peelRef.current && !hasInteracted) {
      const timer = setTimeout(() => {
        peelHint(peelRef.current, {
          peekSize: 25,
          duration: 300,
          repeat: 2,
        });
      }, 1500);
      return () => clearTimeout(timer);
    }
  }, [showHint, hasInteracted]);

  return (
    <PeelWrapper
      ref={peelRef}
      preset="stickyNote"
      height={height}
      width={width}
      drag
      onPeelStart={() => setHasInteracted(true)}
    >
      <PeelTop
        style={{
          backgroundColor: color,
          padding: "16px",
          fontFamily: "'Patrick Hand', cursive, sans-serif",
          fontSize: "16px",
          lineHeight: 1.5,
          boxShadow: "2px 2px 5px rgba(0,0,0,0.1)",
        }}
      >
        {message}
      </PeelTop>
      <PeelBack
        style={{
          backgroundColor: adjustColor(color, -10),
        }}
      />
      <PeelBottom
        style={{
          backgroundColor: "#f5f5f5",
          padding: "16px",
          display: "flex",
          alignItems: "center",
          justifyContent: "center",
          fontStyle: "italic",
          color: "#666",
        }}
      >
        {hiddenMessage}
      </PeelBottom>
    </PeelWrapper>
  );
}

// Helper to darken/lighten colors
function adjustColor(color: string, amount: number): string {
  const hex = color.replace("#", "");
  const num = parseInt(hex, 16);
  const r = Math.min(255, Math.max(0, (num >> 16) + amount));
  const g = Math.min(255, Math.max(0, ((num >> 8) & 0x00ff) + amount));
  const b = Math.min(255, Math.max(0, (num & 0x0000ff) + amount));
  return `#${((r << 16) | (g << 8) | b).toString(16).padStart(6, "0")}`;
}
```

## Styling Tips

1. **Font Choice**: Use a handwritten font like "Patrick Hand" from Google Fonts
2. **Subtle Shadows**: Add a light box-shadow for depth
3. **Slight Rotation**: Rotate notes slightly for a natural look
4. **Color Palette**: Use pastel colors for a classic sticky note feel

```css
/* Optional: Add Google Font */
@import url('https://fonts.googleapis.com/css2?family=Patrick+Hand&display=swap');

/* Rotate for natural look */
.sticky-note-container:nth-child(1) { transform: rotate(-2deg); }
.sticky-note-container:nth-child(2) { transform: rotate(1deg); }
.sticky-note-container:nth-child(3) { transform: rotate(-1deg); }
.sticky-note-container:nth-child(4) { transform: rotate(2deg); }
```
